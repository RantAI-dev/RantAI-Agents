# Mobile App — Phase A (Foundation) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Deliver a runnable Expo iOS+Android app that authenticates against the RantAI-Agents backend (email + password → mobile JWT), supports biometric unlock and multi-account, registers an Expo push token, and renders an empty 5-tab navigation shell — with backend support for mobile auth + push token registration added alongside the existing NextAuth web session (no breaking change for web).

**Architecture:** A new `mobile/` directory holds the Expo SDK 54 / React Native 0.82 app. The existing Next.js backend gains additive routes under `/api/mobile/*` (auth + push) and Prisma models `MobileSession` + `MobilePushToken`. A new universal helper `requireUserAndOrg(req)` accepts EITHER a NextAuth session cookie OR an `Authorization: Bearer <jwt>` header, so every existing dashboard route works for mobile unchanged after a mechanical refactor. Tokens are stored in `expo-secure-store` per-account; biometric unlock gates token release.

**Tech Stack:** Expo SDK 54, React Native 0.82 (New Architecture), React 19 + React Compiler, Expo Router v5, NativeWind v4, react-native-reusables, Reanimated 4, TanStack Query v5, react-native-mmkv, Zustand 5, expo-secure-store, expo-local-authentication, expo-notifications, socket.io-client. Backend: Prisma + PostgreSQL, NextAuth 5, jose (for JWT), bcryptjs (existing), vitest.

**Spec reference:** `docs/superpowers/specs/2026-04-16-mobile-app-react-native-design.md`

### Task 5: Implement `src/lib/auth-mobile.ts`

**Files:**
- Create: `src/lib/auth-mobile.ts`

- [ ] **Step 1: Write the file**

```ts
import { SignJWT, jwtVerify } from "jose"
import { createHash, randomBytes } from "crypto"

interface AccessTokenClaims {
  userId: string
  sessionId: string
}

function getSecret(): Uint8Array {
  const secret = process.env.MOBILE_JWT_SECRET
  if (!secret) throw new Error("MOBILE_JWT_SECRET is not set")
  if (secret.length < 32) throw new Error("MOBILE_JWT_SECRET must be at least 32 characters")
  return new TextEncoder().encode(secret)
}

function getAccessTtlSeconds(): number {
  return Number(process.env.MOBILE_JWT_ACCESS_TTL ?? 900)
}

export function getRefreshTtlSeconds(): number {
  return Number(process.env.MOBILE_JWT_REFRESH_TTL ?? 60 * 60 * 24 * 30)
}

export async function signAccessToken(claims: AccessTokenClaims): Promise<string> {
  const ttl = getAccessTtlSeconds()
  const now = Math.floor(Date.now() / 1000)
  return new SignJWT({ userId: claims.userId, sessionId: claims.sessionId })
    .setProtectedHeader({ alg: "HS256" })
    .setIssuedAt(now)
    .setExpirationTime(now + ttl)
    .setSubject(claims.userId)
    .sign(getSecret())
}

export async function verifyAccessToken(token: string): Promise<AccessTokenClaims> {
  const { payload } = await jwtVerify(token, getSecret(), { algorithms: ["HS256"] })
  if (typeof payload.userId !== "string" || typeof payload.sessionId !== "string") {
    throw new Error("Malformed access token claims")
  }
  return { userId: payload.userId, sessionId: payload.sessionId }
}

export function generateRefreshToken(): string {
  return randomBytes(32).toString("base64url")
}

export function hashRefreshToken(token: string): string {
  return createHash("sha256").update(token).digest("hex")
}
```

- [ ] **Step 1: Write the file**

```ts
import { NextResponse } from "next/server"
import { auth } from "@/lib/auth"
import { prisma } from "@/lib/prisma"
import { verifyAccessToken } from "@/lib/auth-mobile"
import {
  getOrganizationContextWithFallback,
  type OrganizationContext,
} from "@/lib/organization"

export interface AuthorizedRequest {
  user: { id: string }
  orgContext: OrganizationContext
  via: "session" | "mobile"
}

export async function requireUserAndOrg(
  req: Request,
): Promise<AuthorizedRequest | NextResponse> {
  // 1. Try NextAuth session cookie first (web).
  const session = await auth()
  if (session?.user?.id) {
    const orgContext = await getOrganizationContextWithFallback(req, session.user.id)
    if (!orgContext) {
      return NextResponse.json({ error: "No organization context" }, { status: 401 })
    }
    return { user: { id: session.user.id }, orgContext, via: "session" }
  }

  // 2. Fallback: mobile JWT Bearer token.
  const auth_header = req.headers.get("authorization")
  if (auth_header?.startsWith("Bearer ")) {
    const token = auth_header.slice(7).trim()
    try {
      const claims = await verifyAccessToken(token)
      const sess = await prisma.mobileSession.findUnique({
        where: { id: claims.sessionId },
        include: { user: { select: { id: true } } },
      })
      if (!sess || sess.revokedAt || sess.expiresAt < new Date() || sess.userId !== claims.userId) {
        return NextResponse.json({ error: "Invalid session" }, { status: 401 })
      }
      // Update lastUsedAt fire-and-forget — failure must not block the request.
      prisma.mobileSession.update({
        where: { id: sess.id },
        data: { lastUsedAt: new Date() },
      }).catch(() => {})
      const orgContext = await getOrganizationContextWithFallback(req, sess.user.id)
      if (!orgContext) {
        return NextResponse.json({ error: "No organization context" }, { status: 401 })
      }
      return { user: { id: sess.user.id }, orgContext, via: "mobile" }
    } catch {
      return NextResponse.json({ error: "Invalid token" }, { status: 401 })
    }
  }

  return NextResponse.json({ error: "Unauthorized" }, { status: 401 })
}
```

import { NextResponse } from "next/server"
import { z } from "zod"
import { compare } from "bcryptjs"
import { prisma } from "@/lib/prisma"
import {
  signAccessToken,
  generateRefreshToken,
  hashRefreshToken,
  getRefreshTtlSeconds,
} from "@/lib/auth-mobile"

const BodySchema = z.object({
  email: z.string().email(),
  password: z.string().min(1),
  deviceId: z.string().min(1),
  deviceName: z.string().min(1),
  platform: z.enum(["ios", "android"]),
})

export async function POST(req: Request) {
  const json = await req.json().catch(() => null)
  const parsed = BodySchema.safeParse(json)
  if (!parsed.success) {
    return NextResponse.json({ error: "Invalid body" }, { status: 400 })
  }
  const { email, password, deviceId, deviceName, platform } = parsed.data

  const user = await prisma.user.findUnique({ where: { email } })
  if (!user) return NextResponse.json({ error: "Invalid credentials" }, { status: 401 })
  const ok = await compare(password, user.passwordHash)
  if (!ok) return NextResponse.json({ error: "Invalid credentials" }, { status: 401 })

  const refreshToken = generateRefreshToken()
  const refreshTokenHash = hashRefreshToken(refreshToken)
  const expiresAt = new Date(Date.now() + getRefreshTtlSeconds() * 1000)

  const session = await prisma.mobileSession.upsert({
    where: { userId_deviceId: { userId: user.id, deviceId } },
    create: {
      userId: user.id,
      deviceId,
      deviceName,
      platform,
      refreshTokenHash,
      expiresAt,
    },
    update: {
      deviceName,
      platform,
      refreshTokenHash,
      expiresAt,
      lastUsedAt: new Date(),
      revokedAt: null,
    },
  })

  const accessToken = await signAccessToken({ userId: user.id, sessionId: session.id })

  return NextResponse.json({
    accessToken,
    refreshToken,
    user: { id: user.id, email: user.email, name: user.name, role: user.role },
  })
}

import { NextResponse } from "next/server"
import { z } from "zod"
import { prisma } from "@/lib/prisma"
import {
  signAccessToken,
  generateRefreshToken,
  hashRefreshToken,
  getRefreshTtlSeconds,
} from "@/lib/auth-mobile"

const BodySchema = z.object({
  refreshToken: z.string().min(1),
  deviceId: z.string().min(1),
})

export async function POST(req: Request) {
  const json = await req.json().catch(() => null)
  const parsed = BodySchema.safeParse(json)
  if (!parsed.success) {
    return NextResponse.json({ error: "Invalid body" }, { status: 400 })
  }
  const { refreshToken, deviceId } = parsed.data

  const incomingHash = hashRefreshToken(refreshToken)
  const session = await prisma.mobileSession.findUnique({
    where: { refreshTokenHash: incomingHash },
  })
  if (
    !session ||
    session.revokedAt ||
    session.expiresAt < new Date() ||
    session.deviceId !== deviceId
  ) {
    return NextResponse.json({ error: "Invalid refresh token" }, { status: 401 })
  }

  const newRefresh = generateRefreshToken()
  const newHash = hashRefreshToken(newRefresh)
  const newExpiresAt = new Date(Date.now() + getRefreshTtlSeconds() * 1000)

  await prisma.mobileSession.update({
    where: { id: session.id },
    data: {
      refreshTokenHash: newHash,
      expiresAt: newExpiresAt,
      lastUsedAt: new Date(),
    },
  })

  const accessToken = await signAccessToken({
    userId: session.userId,
    sessionId: session.id,
  })

  return NextResponse.json({ accessToken, refreshToken: newRefresh })
}
```

import * as SecureStore from "expo-secure-store"

const INDEX_KEY = "accounts/index"
const ACTIVE_KEY = "accounts/active"

export interface StoredAccount {
  id: string
  email: string
  name: string
  refreshToken: string
}

async function readIndex(): Promise<string[]> {
  const raw = await SecureStore.getItemAsync(INDEX_KEY)
  if (!raw) return []
  try { return JSON.parse(raw) } catch { return [] }
}

async function writeIndex(ids: string[]): Promise<void> {
  await SecureStore.setItemAsync(INDEX_KEY, JSON.stringify(ids))
}

function key(id: string): string {
  return `accounts/${id}`
}

export async function addAccount(acct: StoredAccount): Promise<void> {
  const ids = await readIndex()
  if (!ids.includes(acct.id)) ids.push(acct.id)
  await writeIndex(ids)
  await SecureStore.setItemAsync(key(acct.id), JSON.stringify(acct))
}

export async function listAccounts(): Promise<StoredAccount[]> {
  const ids = await readIndex()
  const out: StoredAccount[] = []
  for (const id of ids) {
    const raw = await SecureStore.getItemAsync(key(id))
    if (raw) {
      try { out.push(JSON.parse(raw)) } catch { /* skip corrupt entry */ }
    }
  }
  return out
}

export async function getAccount(id: string): Promise<StoredAccount | null> {
  const raw = await SecureStore.getItemAsync(key(id))
  if (!raw) return null
  try { return JSON.parse(raw) } catch { return null }
}

export async function removeAccount(id: string): Promise<void> {
  const ids = (await readIndex()).filter(x => x !== id)
  await writeIndex(ids)
  await SecureStore.deleteItemAsync(key(id))
  if (await getActiveAccountId() === id) {
    await SecureStore.deleteItemAsync(ACTIVE_KEY)
  }
}

export async function setActiveAccount(id: string): Promise<void> {
  await SecureStore.setItemAsync(ACTIVE_KEY, id)
}

export async function getActiveAccountId(): Promise<string | null> {
  return SecureStore.getItemAsync(ACTIVE_KEY)
}

export async function clearAllAccounts(): Promise<void> {
  for (const id of await readIndex()) {
    await SecureStore.deleteItemAsync(key(id))
  }
  await SecureStore.deleteItemAsync(INDEX_KEY)
  await SecureStore.deleteItemAsync(ACTIVE_KEY)
}
```

import { env } from "@/lib/env"
import { useAuthStore } from "@/auth/use-auth-store"
import { authApi } from "@/auth/api"
import { getDeviceId } from "@/lib/device-id"

interface RequestOptions extends RequestInit {
  json?: unknown
}

let refreshPromise: Promise<void> | null = null

async function refreshIfNeeded(): Promise<void> {
  if (refreshPromise) return refreshPromise
  refreshPromise = (async () => {
    const { activeAccount, setAccessToken, signOutCurrent } = useAuthStore.getState()
    if (!activeAccount) throw new Error("No active account")
    try {
      const res = await authApi.refresh(activeAccount.refreshToken, getDeviceId())
      await setAccessToken(res.accessToken, res.refreshToken)
    } catch (e) {
      await signOutCurrent()
      throw e
    } finally {
      refreshPromise = null
    }
  })()
  return refreshPromise
}

export async function apiFetch<T = unknown>(
  path: string,
  options: RequestOptions = {},
): Promise<T> {
  const { accessToken } = useAuthStore.getState()
  const headers = new Headers(options.headers)
  if (accessToken) headers.set("Authorization", `Bearer ${accessToken}`)
  if (options.json !== undefined) {
    headers.set("Content-Type", "application/json")
    options.body = JSON.stringify(options.json)
  }
  const res = await fetch(`${env.apiBaseUrl}${path}`, { ...options, headers })
  if (res.status === 401 && accessToken) {
    await refreshIfNeeded()
    return apiFetch<T>(path, options) // retry once
  }
  if (!res.ok) {
    throw Object.assign(new Error(`${path} ${res.status}`), { status: res.status })
  }
  if (res.status === 204) return undefined as T
  return res.json() as Promise<T>
}

  addAccount,

  listAccounts,
  removeAccount,
  setActiveAccount,
  type StoredAccount,

import { authApi } from "./api"

interface AuthState {
  ready: boolean
  accessToken: string | null
  activeAccount: StoredAccount | null
  setAccessToken: (access: string, newRefresh?: string) => Promise<void>
  bootstrap: () => Promise<void>
  signIn: (email: string, password: string) => Promise<void>
  switchAccount: (id: string) => Promise<void>
  signOutCurrent: () => Promise<void>
}

export const useAuthStore = create<AuthState>((set, get) => ({
  ready: false,
  accessToken: null,
  activeAccount: null,

  async setAccessToken(access, newRefresh) {
    const current = get().activeAccount
    if (current && newRefresh) {
      const updated = { ...current, refreshToken: newRefresh }
      await addAccount(updated)
      set({ activeAccount: updated })
    }
    set({ accessToken: access })
  },

  async bootstrap() {
    const id = await getActiveAccountId()
    if (!id) {
      set({ ready: true })
      return
    }
    const acct = await getAccount(id)
    set({ activeAccount: acct, ready: true })
  },

  async signIn(email, password) {
    const platform = (await import("react-native")).Platform.OS as "ios" | "android"
    const { getDeviceId } = await import("@/lib/device-id")
    const deviceId = getDeviceId()
    const deviceName = Device.deviceName ?? Device.modelName ?? "Mobile"
    const res = await authApi.login({ email, password, deviceId, deviceName, platform })
    const acct: StoredAccount = {
      id: res.user.id,
      email: res.user.email,
      name: res.user.name,
      refreshToken: res.refreshToken,
    }
    await addAccount(acct)
    await setActiveAccount(acct.id)
    set({ activeAccount: acct, accessToken: res.accessToken })
  },

  async switchAccount(id) {
    const acct = await getAccount(id)
    if (!acct) throw new Error("Account not found")
    await setActiveAccount(id)
    set({ activeAccount: acct, accessToken: null })
    // Trigger an immediate refresh on next API call.
  },

  async signOutCurrent() {
    const current = get().activeAccount
    if (!current) return
    try { await authApi.logout(current.refreshToken) } catch { /* network failure is fine */ }
    await removeAccount(current.id)
    const remaining = await listAccounts()
    if (remaining[0]) {
      await setActiveAccount(remaining[0].id)
      set({ activeAccount: remaining[0], accessToken: null })
    } else {
      set({ activeAccount: null, accessToken: null })
    }

  if (!Device.isDevice) return null // simulators can't get tokens
  const { status: existing } = await Notifications.getPermissionsAsync()
  let final = existing
  if (existing !== "granted") {
    const ask = await Notifications.requestPermissionsAsync()
    final = ask.status
  }
  if (final !== "granted") return null

  const tokenData = await Notifications.getExpoPushTokenAsync()
  const expoPushToken = tokenData.data
  const platform = Platform.OS === "ios" ? "ios" : "android"
  await apiFetch("/api/mobile/push/register", {
    method: "POST",
    json: { deviceId: getDeviceId(), expoPushToken, platform },
  })
  return expoPushToken
}

import { useEffect } from "react"
import { useRouter, useSegments } from "expo-router"
import { useAuthStore } from "@/auth/use-auth-store"
import {
  isBiometricEnabled,
  needsBiometricUnlock,
} from "@/auth/biometric"

export function AuthGate({ children }: { children: React.ReactNode }) {
  const ready = useAuthStore(s => s.ready)
  const activeAccount = useAuthStore(s => s.activeAccount)
  const accessToken = useAuthStore(s => s.accessToken)
  const segments = useSegments()
  const router = useRouter()

  useEffect(() => {
    if (!ready) return
    const inAuthGroup = segments[0] === "(auth)"
    if (!activeAccount && !inAuthGroup) {
      router.replace("/(auth)/login")
      return
    }
    if (activeAccount && isBiometricEnabled() && needsBiometricUnlock() && segments.join("/") !== "(auth)/biometric") {
      router.replace("/(auth)/biometric")
      return
    }
    if (activeAccount && !isBiometricEnabled() && inAuthGroup && segments[1] !== "account-picker") {
      router.replace("/(tabs)/chat")
    }
  }, [ready, activeAccount, accessToken, segments, router])

  return <>{children}</>
}
